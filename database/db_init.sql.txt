CREATE DATABASE gestion_practicas;
USE gestion_practicas;

CREATE TABLE usuarios (
  id_usuario INT AUTO_INCREMENT PRIMARY KEY,
  nombre VARCHAR(100) NOT NULL,
  correo VARCHAR(100) UNIQUE NOT NULL,
  contraseña VARCHAR(255) NOT NULL,
  rol ENUM('estudiante','tutor','empresa','admin') NOT NULL
);

CREATE TABLE proyectos (
  id_proyecto INT AUTO_INCREMENT PRIMARY KEY,
  nombre VARCHAR(100) NOT NULL,
  descripcion TEXT,
  empresa_id INT,
  tutor_id INT,
  fecha_inicio DATE,
  fecha_fin DATE,
  FOREIGN KEY (empresa_id) REFERENCES usuarios(id_usuario),
  FOREIGN KEY (tutor_id) REFERENCES usuarios(id_usuario)
);

CREATE TABLE asignaciones (
  id_asignacion INT AUTO_INCREMENT PRIMARY KEY,
  proyecto_id INT,
  estudiante_id INT,
  fecha_inicio DATE,
  fecha_fin DATE,
  FOREIGN KEY (proyecto_id) REFERENCES proyectos(id_proyecto),
  FOREIGN KEY (estudiante_id) REFERENCES usuarios(id_usuario)
);

CREATE TABLE evidencias (
  id_evidencia INT AUTO_INCREMENT PRIMARY KEY,
  asignacion_id INT,
  descripcion TEXT,
  archivo VARCHAR(255),
  fecha DATE,
  FOREIGN KEY (asignacion_id) REFERENCES asignaciones(id_asignacion)
);

CREATE TABLE reportes (
  id_reporte INT AUTO_INCREMENT PRIMARY KEY,
  proyecto_id INT,
  contenido TEXT,
  fecha DATE,
  FOREIGN KEY (proyecto_id) REFERENCES proyectos(id_proyecto)
);
